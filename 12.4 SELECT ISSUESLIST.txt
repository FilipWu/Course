SELECT A.*, B.NAME
FROM ISSUES A
JOIN ISSUESLISTS B ON A.ISSUESLIST_ID = B.ID;


SELECT A.ID TASK_ID, A.ISSUESLIST_ID, A.SUMMARY, A.DESCRIPTION, B.ID USER_ID, B.FIRSTNAME, B.LASTNAME
FROM ISSUES A
         JOIN USERS B ON A.USER_ID_ASSIGNEDTO = B.ID;


SELECT B.FIRSTNAME, B.LASTNAME, COUNT(A.ID) AS TASK_COUNT
FROM USERS B
LEFT JOIN ISSUES A ON A.USER_ID_ASSIGNEDTO = B.ID
GROUP BY B.FIRSTNAME, B.LASTNAME;
